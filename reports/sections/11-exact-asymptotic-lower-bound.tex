\section{Derivation of the Recurrence}

Recall that our shifted yield recurrence is
$$Y_{n+1} \stackrel{d}{=} I^{(n)}\left(Y_{U_n}^{(1)} + Y_{n-1-U_n}^{(2)} + 1\right) + J^{(n)}\max\left(Y_{U_n}^{(1)}, Y_{n-1-U_n}^{(2)}\right), \quad Y_1=Y_2=0.$$
Taking expectation on both sides, and letting $y_n=\EE[Y_n]$, we have
$$y_{n+1} = \dfrac{2u}{n}\sum_{i=0}^{n-1}y_i + \dfrac{v}{n}\sum_{i=0}^{n-1}\EE\left[\max\left(Y_{i}^{(1)}, Y_{n-1-i}^{(2)}\right)\right]+u, \quad n\ge 1, \quad y_0=y_1=0.$$

Since $\max(Y_i, Y_{n-1-i}) \ge Y_i$ and $\max(Y_i, Y_{n-1-i}) \ge Y_{n-1-i}$, we have
$$\EE[\max(Y_i, Y_{n-1-i})] \ge \max(y_i, y_{n-1-i}).$$
Hence, for every $n\ge 1$, we have
$$y_{n+1} \ge \dfrac{2u}{n}\sum_{i=0}^{n-1}y_i + \dfrac{v}{n}\sum_{i=0}^{n-1}\max(y_{i}, y_{n-1-i})+u.$$

Define $(y'_n)$ such that $y'_0=y'_1=0$ and for every $n\ge 1$,
$$y'_{n+1} = \dfrac{2u}{n}\sum_{i=0}^{n-1}y'_i + \dfrac{v}{n}\sum_{i=0}^{n-1}\max(y'_{i}, y'_{n-1-i})+u.$$
Then $y_n \ge y'_n$ for every $n\ge 0$. Finally, we homogenize the recurrence by defining
$$x_n = \dfrac{2u+v-1}{u}y'_n + 1$$
for every $n\ge 0$. We have $x_0=x_1=1$ and for every $n\ge 1$. The recurrence of $(x_n)$ is
\begin{equation}
  \label{eq:recurrence-sum-diff}
  x_{n+1} = \dfrac{2u}{n}\sum_{i=0}^{n-1}x_i + \dfrac{v}{n}\sum_{i=0}^{n-1}\max(x_{i}, x_{n-1-i}), \quad n \ge 1, \quad  x_0=x_1=1.
\end{equation}
We consider the case where $2u+v>1$.

\section{Analysis of the Recurrence}

Throughout this section, let $\alpha$ be the unique solution of the equation
\begin{equation}
  \alpha + v2^{-\alpha} = 2(u+v)-1.
\end{equation}

Note that $\alpha>0$ since otherwise, we would have $2u+v = 1$.

\begin{proposition}
  The sequence $(x_n)$ is increasing.
\end{proposition}

\begin{proof}
  We have $x_0=x_1=1$ and $x_2=2u+v$. Assume that $x_{i+1}\ge x_i$ for all $0\le i \le n-1$ with $n\ge 2$. We have
  \begin{align*}
    nx_{n+1}
     & = 2u\sum_{i=0}^{n-1}x_i + v\sum_{i=0}^{n-1}\max(x_{i}, x_{n-1-i})                                       \\
     & = 2u\sum_{i=0}^{n-1}x_i + 2v\sum_{i=\lfloor (n+1)/2\rfloor}^{n-1}x_{i}+v\delta_nx_{\lfloor n/2\rfloor}, \\
  \end{align*}
  where $\delta_n = 1$ if $n$ is odd and $\delta_n = 0$ if $n$ is even.

  Let us compute $nx_{n+1}-(n-1)x_n$ for $n\ge 2$. Let $k\ge 1$ be an integer.
  \begin{itemize}
    \item If $n=2k$, we have
          $$2kx_{2k+1}-(2k-1)x_{2k} = 2(u+v)x_{2k-1} - vx_{k-1}.$$
    \item If $n=2k+1$, we have
          $$(2k+1)x_{2k+2}-2kx_{2k+1} = 2(u+v)x_{2k} -2vx_{k} + vx_{k} = 2(u+v)x_{2k} - vx_{k}.$$
  \end{itemize}
  Collecting two cases, we have
  \begin{align*}
         & nx_{n+1}-(n-1)x_n = 2(u+v)x_{n-1} - vx_{\lfloor (n-1)/2\rfloor}    \\
    \iff & n(x_{n+1}-x_n) = 2(u+v)x_{n-1} - x_n - vx_{\lfloor (n-1)/2\rfloor} \\
  \end{align*}
  We have
  $$x_n=\dfrac{2u}{n}\sum_{i=0}^{n-2}x_i+\dfrac{v}{n}\sum_{i=0}^{n-2}\max(x_{i}, x_{n-2-i}) \le (2u+v)x_{n-2}.$$
  Hence,
  \begin{align*}
    n(x_{n+1}-x_n)
     & \ge 2(u+v)x_{n-1} - (2u+v)x_{n-2} - vx_{\lfloor (n-1)/2\rfloor} \\
     & \ge 2(u+v)x_{n-1} - (2u+v)x_{n-1} - vx_{n-1}                    \\
     & \ge 0.
  \end{align*}

\end{proof}

\begin{proposition}
  The sequence $(x_n)$ satisfies the recurrence
  \begin{equation}
    nx_{n+1}-(n-1)x_n = 2(u+v)x_{n-1} - vx_{\lfloor (n-1)/2\rfloor}, \quad n\ge 1,
  \end{equation}
  with $x_0=x_1=1$.
\end{proposition}



\begin{lemma}
  For every $n\ge0$, we have
  $$M_n = \sum_{i=0}^{n-1} \max(i^\alpha, (n-1-i)^\alpha) \le 2\int_{n/2}^n x^\alpha dx.$$
\end{lemma}

\begin{proof}
  Let $n=2k$. We have
  $$M_{2k} = 2 \sum_{i=k}^{2k-1} i^\alpha \le 2 \int_k^{2k} x^\alpha dx = 2 \int_{n/2}^n x^\alpha dx.$$
  Let $n=2k+1$. We have $k^\alpha = 2 \times \frac{1}{2} \times k^\alpha \le 2 \int_{k+1/2}^{k+1} x^\alpha dx$. Hence,
  $$M_{2k+1} \le 2 \int_{k+1/2}^{k+1} x^\alpha dx + 2 \int_{k+1}^{2k+1} x^\alpha dx = 2 \int_{k+1/2}^{2k+1} x^\alpha dx = 2 \int_{n/2}^n x^\alpha dx.$$
\end{proof}

\begin{lemma}
  For every $n\ge1$, we have $x_n \le Kn^\alpha$ for some $K>0$.
\end{lemma}
\begin{proof}
  We have $x_2=2u+v$. Assume that $x_i \le Ki^\alpha$ for all $1\le i \le n$ with $n\ge 2$ and $K>0$ is to be determined. For $n\ge 2$, we have
  \begin{align*}
    nx_{n+1}
     & = 2u + 2u\sum_{i=1}^{n-1}x_i + v\sum_{i=0}^{n-1}\max(x_i, x_{n-1-i})                                                        \\
     & \le 2u + 2uK\int_{1}^{n}x^\alpha dx + 2vK\int_{n/2}^{n}x^\alpha dx                                                          \\
     & = 2u + \dfrac{2uK}{\alpha+1}\left(n^{\alpha+1}-1\right) + \dfrac{2vK}{\alpha+1}\left(n^{\alpha+1} - (n/2)^{\alpha+1}\right) \\
     & = 2u + \dfrac{K n^{\alpha+1}}{\alpha+1}\left(2u + 2v\left(1 - 2^{-\alpha-1}\right)\right)-\dfrac{2uK}{\alpha+1}             \\
     & = 2u + K n^{\alpha+1} -\dfrac{2uK}{\alpha+1}.
  \end{align*}
  Note that in the sum of maxima, we do not have to isolate $\max(x_0, x_{n-1})$, since it is automatically evaluated as $x_{n-1}$, which is already included in the integral. Choose $K=\alpha+1$, we check that the base cases hold
  \begin{align*}
     & \alpha + 1 \ge 1,                                                                             \\
     & (\alpha+1)2^\alpha = \left(2(u+v)-v2^{-\alpha}\right)2^\alpha = 2^{\alpha+1}(u+v)-v \ge 2u+v.
  \end{align*}

  Therefore,
  $$x_{n+1} \le Kn^\alpha \le K(n+1)^\alpha.$$
\end{proof}

\begin{theorem}
  We have $x_n \sim Cn^\alpha$ for some $C>0$ to be determined and $\alpha$ is the unique solution of the equation
  $$\alpha + v2^{-\alpha} = 2(u+v)-1$$
\end{theorem}

\begin{proof}
  Let $z_n=\dfrac{x_n}{(n-1)^\alpha}$ for $n\ge 2$. We will prove that $(z_n)$ converges by proving that $\sum_{n\ge 0}|z_{n+1}-z_n|$ converges. We have, for $n\ge 5$,
  \begin{align*}
         & n^{\alpha+1}z_{n+1} - (n-1)^{\alpha+1}z_n = 2(u+v)(n-2)^{\alpha}z_{n-1} - v\left(\left\lfloor \dfrac{n-1}{2}\right\rfloor-1\right)^{\alpha}z_{\lfloor (n-1)/2\rfloor}                                                 \\
    \iff & z_{n+1} = \left(1-\dfrac{1}{n}\right)^{\alpha+1}z_n + \dfrac{2(u+v)}{n}\left(1-\dfrac{2}{n}\right)^{\alpha}z_{n-1} - \dfrac{v}{n}\left(\dfrac{\lfloor (n-1)/2\rfloor-1}{n}\right)^{\alpha}z_{\lfloor (n-1)/2\rfloor}.
  \end{align*}
  We have
  \begin{align*}
     & \left(1-\dfrac{1}{n}\right)^{\alpha+1} = 1 - \dfrac{\alpha+1}{n} + O\left(\dfrac{1}{n^2}\right),                                                                                                  \\
     & \dfrac{1}{n}\left(1-\dfrac{2}{n}\right)^{\alpha} = \dfrac{1}{n} + O\left(\dfrac{1}{n^2}\right),                                                                                                   \\
     & -\frac{1}{n}\left(\dfrac{\lfloor (n-1)/2\rfloor-1}{n}\right)^{\alpha}=-\frac{1}{n} \left( 2^{-\alpha} + O\left(\frac{1}{n}\right) \right) = -\frac{2^{-\alpha}}{n} + O\left(\frac{1}{n^2}\right).
  \end{align*}
  By the previous lemma, $(z_n)$ is bounded. Therefore,
  \begin{align*}
    z_{n+1}-z_n
     & = -\dfrac{\alpha+1}{n}z_n + \dfrac{2(u+v)}{n}z_{n-1} - \frac{v2^{-\alpha}}{n}z_{\lfloor (n-1)/2\rfloor} + O\left(\frac{1}{n^2}\right) \\
     & = -\dfrac{2(u+v)}{n}(z_n-z_{n-1}) + \frac{v2^{-\alpha}}{n}(z_{n}-z_{\lfloor (n-1)/2\rfloor}) + O\left(\frac{1}{n^2}\right).
  \end{align*}

  Let $\nabla z_n = z_{n} - z_{n-1}$. Then, there exists a constant $L>0$ such that for every $n\ge 5$,
  $$|\nabla z_{n+1}| \le \dfrac{2(u+v)}{n}|\nabla z_{n}| + \frac{v2^{-\alpha}}{n}\sum_{i=\lfloor (n+1)/2\rfloor}^{n}|\nabla z_{i}| + \dfrac{L}{(n+1)^2}.$$
  Equivalently, for every $n\ge 6$,
  $$|\nabla z_n| \le \dfrac{2(u+v)}{n-1}|\nabla z_{n-1}| + \frac{v2^{-\alpha}}{n-1}\sum_{i=\lfloor n/2\rfloor}^{n-1}|\nabla z_{i}| + \dfrac{L}{n^2}.$$

  Let $u_m = \max_{2^m \le n < 2^{m+1}} |\nabla z_n|$. Then, for every $n\in [2^m, 2^{m+1})$, we have
  \begin{align*}
     & \dfrac{1}{n-1}|\nabla z_{n-1}| \le \dfrac{1}{n-1}\max_{2^m-1\le i < 2^{m+1}-2}|\nabla z_i| = \dfrac{1}{n-1}\max(u_{m-1},u_m),                                                                      \\
     & \dfrac{1}{n-1}\sum_{i=\lfloor n/2\rfloor}^{n-1}|\nabla z_{i}| \le \dfrac{n+1}{2(n-1)}\max_{2^{m-1}\le i < 2^{m+1}} |\nabla z_i| \le \left(\dfrac{1}{2}+\dfrac{1}{n-1}\right) \max(u_{m-1}, u_{m}).
  \end{align*}
  Therefore,
  \begin{align*}
    |\nabla z_{n}|
     & \le \dfrac{2(u+v)}{n-1}\max(u_{m-1},u_m) + v2^{-\alpha}\left(\dfrac{1}{2}+\dfrac{1}{n-1}\right)\max(u_{m-1}, u_m) + \dfrac{L}{2^{2m}} \\
     & = \left(\dfrac{\alpha+1}{n-1}+ v2^{-\alpha-1}\right)\max(u_{m-1}, u_m) + \dfrac{L}{2^{2m}}.
  \end{align*}
  Since $v2^{-\alpha} = 2(u+v)-\alpha-1 < 1$, we can choose $m_0$ large enough such that for every $m\ge m_0$ and $n\in [2^m, 2^{m+1})$, there exists $\gamma_1$ such that $\dfrac{\alpha+1}{n-1}+ v2^{-\alpha-1} < \gamma_1 < \dfrac{1}{2}$. Taking the maximum over $n\in [2^m, 2^{m+1})$, we have
  $$u_m \le \gamma_1\max(u_{m-1}, u_m) + \dfrac{L}{2^{2m}}.$$

  If $u_{m-1} > u_m$, then $u_m \le \gamma_1 u_{m-1} + \dfrac{L}{2^{2m}}.$

  If $u_{m-1} \le u_m$, then $u_m \le \dfrac{L}{2^{2m}(1-\gamma_1)}.$

  In both cases, we have
  \begin{align*}
    u_m
     & \le \gamma_1 u_{m-1} + \dfrac{L}{2^{2m}(1-\gamma_1)}                                                             \\
     & \le \sum_{i=m_0+1}^m \gamma_1^{m-i}\dfrac{L}{2^{2i}(1-\gamma_1)} + \gamma_1^{m-m_0}u_{m_0}                       \\
     & < \dfrac{L}{(1-\gamma_1)}\sum_{i=m_0+1}^m \gamma_1^{m-i}\left(\dfrac{1}{4}\right)^{i} + \gamma_1^{m-m_0}u_{m_0}.
  \end{align*}
  Choose $\gamma_2$ such that $\max\left(\gamma_1, \dfrac{1}{4}\right) < \gamma_2 < \dfrac{1}{2}$. We have
  \begin{align*}
    u_m < \dfrac{L}{(1-\gamma_1)}\sum_{i=m_0+1}^m \gamma_2^m + \gamma_2^{m-m_0}u_{m_0} \le L'm\gamma_2^m,
  \end{align*}
  for some constant $L'>0$. Choose $\gamma_3$ such that $\gamma_2 < \gamma_3 < \dfrac{1}{2}$. Then there exists $m_1>m_0$ such that for every $m\ge m_1$, we have $L'm\gamma_2^m < L'\gamma_3^m$, implying that,
  $$u_m < L'\gamma_3^m.$$
  Therefore,
  \begin{align*}
    \sum_{n=0}^\infty|\nabla z_{n}| = \sum_{m=0}^\infty \sum_{n=2^m}^{2^{m+1}-1}|\nabla z_n| = \sum_{m=0}^\infty 2^m u_m < \sum_{m=0}^{m_1-1} 2^m u_m  + L'\sum_{m=m_1}^\infty (2\gamma_3)^{m} < \infty.
  \end{align*}
  Hence, $(z_n)$ converges to some constant $C>0$. Thus, $x_n \sim Cn^\alpha$.
\end{proof}
